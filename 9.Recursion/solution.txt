class Solution {
    public double myPow(double x, int n) {
        double ans = 1.0;
        long nn = n;//change n to nn to avoid integer overflow for int absolute value
        if (nn < 0) nn = nn * -1; //conver negative value to positive
        while (nn > 0) {
            //odd nn
            if (nn % 2 == 1) {
                ans = ans * x;
                nn = nn - 1;
            } else {
                x = x * x;
                nn = nn / 2;
            }
        }
        if (n < 0)
            return (double) 1.0 / (double) ans;
        return ans;
    }
}

9002. class Solution {
    
    static final long MOD = 1_000_000_007;

    // Fast exponentiation function
    public long power(long base, long exponent) {
        long result = 1;
        base = base % MOD;

        while (exponent > 0) {
            if (exponent % 2 == 1) {
                result = (result * base) % MOD;
            }
            base = (base * base) % MOD;
            //Optimize exponentiation
            exponent /= 2;
        }

        return result;
    }

    public int countGoodNumbers(long n) {
        long evenCount = (n + 1) / 2; // ceil(n / 2)
        long oddCount = n / 2;        // floor(n / 2)

        long evenChoices = power(5, evenCount);
        long oddChoices = power(4, oddCount);

        return (int)((evenChoices * oddChoices) % MOD);
    }

}

   class Solution {
    public List<String> generateParenthesis(int n) {

        List<String> result = new ArrayList<>();
        backtrack(n, n, result, "");
        return result;
    }

    void backtrack(int openN, int closeN, List<String> result, String current) {
        //Base case
        if (openN == 0 && closeN == 0) {
            result.add(current);
            return;
        }
        // Add '(' if any are left
        if (openN>0) {
            backtrack(openN - 1, closeN, result, current + "(" );
        }
        //Add ')' only if it won't break the balance
        if (closeN>openN) {
            backtrack( openN, closeN - 1, result, current + ")" );
        }
    }
}

            return -1;

        if (arr[m] == target) {
            return m;
        }
        if (target < arr[m]) {
            return binaryRec(arr, target, s, m - 1);
        }
        return binaryRec(arr, target, m + 1, e);
    }

       int sortedCheck(int[] arr, int s, int e, int target) {
        //Base case as couldn't find target
        if (s > e) {
            return -1;
        }
        //if mid element is the target element
        int m = s + (e - s) / 2;
        if (arr[m] == target) {
            return m;
        }
        //if mid is greater than start(left half is sorted) -->element range is fixed here
        if (arr[m] >= arr[s]) {
            if (target >= arr[s] && target <= arr[m]) {
                return sortedCheck(arr, s, m - 1, target);
            } else {
                return sortedCheck(arr, m + 1, arr.length - 1, target);
            }
        }
        //if mid is smaller than end (right half is sorted) -->element range is fixed here
        else {
            if (target >= arr[m] && target <= arr[e]) {
                return sortedCheck(arr, m + 1, e, target);
            }
            return sortedCheck(arr, s, m - 1, target);
        }
    }

    public class BubbleSortRec {
    public static void main(String[] args) {
        int[] arr = {1, 3, 4, 2};
        sort(arr, arr.length - 1, 0);
        System.out.println(Arrays.toString(arr));
    }

    private static void sort(int[] arr, int length, int count) {
        //Base case--> when length became 0 just return
        if (length == 0) {
            return;
        }
        /*if count is smaller than length keep on comparing and swap if required
         to push the largest number to end*/
        if (count < length) {
            //check if next element is smaller
            if (arr[count] > arr[count + 1]) {
                //if yes swap
                int temp = arr[count];
                arr[count] = arr[count + 1];
                arr[count + 1] = temp;
            }
            //after swap check same element to the other next elements
            sort(arr, length, count + 1);
        }
        //after one iteration is done reduce the comparison size by 1
        sort(arr, length - 1, 0);
    }
}

    }

     static void sort(int[] arr, int r, int c,int maxIdx) {
        //Base case
        if(r==0){
            return;
        }
        //Make element comparison from c(Current left index) to r(end index)
        if(c<r){
            if(arr[c]>arr[maxIdx]){
                sort(arr,r,c+1,c);
            }else{
                sort(arr,r,c+1,maxIdx);
            }
        }else{
            int temp=arr[maxIdx];
            arr[maxIdx]=arr[r-1];
            arr[r-1]=temp;
        }
        sort(arr,r-1,0,0);

    }
}

       int[] sort(int[] arr) {
        //Base case
        if (arr.length == 1) {
            return arr;
        }
        int mid = arr.length / 2;
        //sort first part of the array
        int[] first = sort(Arrays.copyOfRange(arr, 0, mid));
        //sort second part of the array
        int[] last = sort(Arrays.copyOfRange(arr, mid, arr.length));
        //merge sorted parts after sorting them while merging
        return merge(first, last);

    }

     int[] merge(int[] first, int[] last) {
        int[] mix = new int[first.length + last.length];
        int i = 0, j = 0, k = 0;
        while (i < first.length && j < last.length) {
            //first[i]<last[j]
            if (first[i] < last[j]) {
                mix[k] = first[i];
                i++;
            } else {
                mix[k] = last[j];
                j++;
            }
            k++;
        }
        while(i<first.length){
            mix[k]=first[i];
            k++;
            i++;
        }
        while(j<last.length){
            mix[k]=last[j];
            k++;
            j++;
        }
        return mix;
    }

        void sortInPlace(int[] arr,int s,int e) {
        //Base case(Means when just one element is left)
        if (e-s == 1) {
            return;
        }
        int mid = (s+e)/2;
        //sort first part of the array
        sortInPlace(arr, s, mid);
        //sort second part of the array
         sortInPlace(arr, mid, e);
        //merge sorted parts after sorting them while merging
         mergeInPlace(arr,s,mid,e);

    }

      static void mergeInPlace(int[] arr,int s,int mid,int e) {
        //Create a result array that will hold sorted resulted array
        int[] mix = new int[e-s];
        int i=s;
        int j=mid;
        int k=0; //counter var for index of result array
        while (i < mid && j < e) {
            //arr[i]<arr[j]
            if (arr[i] < arr[j]) {
                mix[k] = arr[i];
                i++;
            } else {
                mix[k] = arr[j];
                j++;
            }
            k++;
        }
        //If any sub part of the array length is less and some elements are still left for comparision
        while(i<mid){
            mix[k]=arr[i];
              k++;
            i++;
        }
        while(j<e){
            mix[k]=arr[j];
            k++;
            j++;
        }
        //Populate the result array in main array
        for(int l=0;l<mix.length;l++){
            arr[s+l]=mix[l];
        }
    }

    public class QuickSort {
    public static void main(String[] args) {
        int[] num = {5, 4, 3, 2, 1};
        sort(num, 0, num.length - 1);
        System.out.println(Arrays.toString(num));
    }

    //Here low is left most index, hi is right most index
    private static void sort(int[] num, int low, int hi) {
        //Base case: left crosses right index
        if (low >= hi) {
            return;
        }
        int s = low; //current pointer of low set to s
        int e = hi;// current pointer hi set as e
        int m = s + (e - s) / 2;
        int pivot = num[m]; //chosen mid as pivot

        while (s <= e) {
            //No violation cases result in moving indexes
            while (num[s] < pivot) {
                s++;
            }
            while (num[e] > pivot) {
                e--;
            }
            //if violated swap the violating case
            if (s <= e) {
                int temp = num[s];
                num[s] = num[e];
                num[e] = temp;
                s++;
                e--;
            }
        }
        //recursive call we will have two sub arrays as
                //Recursive call is as per while (s <= e) condition which results in sub array as "low to e" && "s to h".
        sort(num, low, e);
        sort(num, s, hi);

    }
    }

                if (r > 1 && c > 1) {
                    mazeDiagonal(p + "D", r - 1, c-1);
                }
                //Only go down when you have row left
                if (r > 1) {
                    mazeDiagonal(p + "V", r - 1, c);
                }
                //Only go down when you have column left --> left
                if (c > 1) {
                    mazeDiagonal(p + "H", r, c - 1);
                }

            }
        }

    package Recursion;

    public class MazeWithObstacle {
        public static void main(String[] args) {
            //false one is obstacle
            boolean[][] board = {
                    {true, true, true},
                    {true, false, true},
                    {true, true, true}
            };
            //p is processed result
            mazeObst("", board, 0, 0);
        }

        static void mazeObst(String p, boolean[][] board, int r, int c) {
            //if row is exhausted or column is exhausted
            if (r == board.length - 1 && c == board[0].length - 1) {
                System.out.println(p);
                return;
            }
            //if you are at obstacle return (No recursion calls of that branch
            if (!board[r][c]) {
                return;
            }
            //can go down if row is not exhausted
            if (r < board.length - 1) {
                mazeObst(p + "D", board, r + 1, c);
            }
            //can go right if column is not exhausted
            if (c < board[0].length - 1) {
                mazeObst(p + "R", board, r, c + 1);
            }

        }
    }

        //Base case

        if(index==input.length()){
            System.out.println(current);
            return;
        }
        //induction

public class RecSortArray {
    public static void main(String[] args) {
        int[] arr = {2, 0, 1, 4};
        int n = arr.length;
        sort(arr, n);
        System.out.println(Arrays.toString(arr));
    }

    private static void sort(int[] arr, int n) {
        //base case{when n is zero
        if (n <= 1) {
            return;
        }
//        induction
        //bring out last element
        int temp = arr[n-1];
        //sort smaller array
        sort(arr, n-1);
        //insert the brought out value back in array
        insertA(arr, temp, n-1);
    }

    private static void insertA(int[] arr, int temp, int n) {
        //BASE CASE{when n is zero or arr[n] is smaller than temp , insert the value at last
        if (n == 0 || arr[n - 1] <= temp) {
            arr[n] = temp;
        } else {
             //shift element to right
            arr[n]=arr[n-1];
            // recurse insert over sorter array
            insertA(arr, temp, n - 1);
        }

    }

public class RecRevStack {
    public static void main(String[] args) {
        Stack<Integer> s = new Stack<>();
        s.push(0);
        s.push(3);
        s.push(1);
        s.push(2);
        System.out.println(s.toString());
        rev(s);
        System.out.println(s.toString());
    }

    private static void rev(Stack<Integer> s) {
        //base case{when n is zero
        if (s.size() <= 1) {
            return;
        }
//        induction
        //bring out last element
        int temp = s.pop();
        //sort smaller array
        rev(s);
        //insert the brought out value back in array
        insertA(s, temp);
    }

    private static void insertA(Stack<Integer> s, int temp) {
        //BASE CASE{when n is zero or arr[n] is smaller than temp , insert the value at last
        if (s.isEmpty()) {
            s.push(temp);
            return;
        } else {
            //shift element to right
            int value=s.pop();
            // recurse insert over sorter array
            insertA(s, temp);
            s.push(value);
        }

    }

}

public class RecDelMidOfStack {
    public static void main(String[] args) {
        Stack<Integer> s = new Stack<>();
        s.push(0);
        s.push(3);
        s.push(2);
        s.push(8);
        System.out.println(s.size());
        System.out.println(s.toString());
        sort(s);
        System.out.println(s.toString());
    }

    private static void sort(Stack<Integer> s) {
        //base case , return when stack became empty
        if (s.isEmpty()) {
            return;
        }
        //mid of stack with safe odd even
        int k=s.size()/2+1;
//        induction
        //solve the deletion of mid-element
        solve(s, k);
    }

    private static void solve(Stack<Integer> s, int mid) {
        //BASE CASE{when k is at the element which we want to delete, just pop that element
        if (mid==1) {
            s.pop();
            return;
        } else {
            //pop the top element and pass the smaller input
           int value=s.pop();
           solve(s,mid-1);
           //push back the popped element
           s.push(value);
           return;
        }

    }

}

public class RecNBitBinary {
    public static void main(String[] args) {
        binary(3);
    }

    static void binary(int n) {
        //ou can't choose close before open so we can start with open and then choose either open or close
        String op = "1";
        helper(n, 0, 1, op);
    }

    static void helper(int n, int zero, int ones, String op) {
        //Base case
        //when op is an n bit number
        if (op.length() == n) {
            System.out.println(op);
            return;
        }
        //induction
        //1 can be always choose if the length of output is still in range of n , as number 1 can be any times in op
        //but you can choose 0 only when your zeros are less than ones

        helper(n, zero, ones + 1, op + "1");

        if (zero < ones) {
            helper(n, zero + 1, ones, op + "0");
        }
        return;
    }
}

public class RecCasePermutation {
    public static void main(String[] args) {
        String s = "a1c";
        solve(s, "");
    }

    private static void solve(String ip, String op) {
        //base case , return when stack became empty
        if (ip.isEmpty()) {
            System.out.println(op);
            return;
        }
         //induction
        //bring out first element
        // small the input
        //if it is a letter then we have two options to add it in output, small or capital
        char ch = ip.charAt(0);
        String rest = ip.substring(1);
        if (Character.isLetter(ch)) {
            //hypothesis
            solve(rest, op + Character.toLowerCase(ch));
            solve(rest, op + Character.toUpperCase(ch));
        } else {
            //if it is a digit then we have only one option to add it in output
            solve(rest, op + ch);
        }
        return;
    }
